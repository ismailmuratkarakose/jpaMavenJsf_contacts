Bu bir Maven-Jpa frameworkleri kullanilarak hazirlanmiþ Jsf(Primefaces) arayüzlü bir programdir.
Bu uygulama maven araciligiyla kütüphaneleri daha kolay eriþmek için kullandim. Ayni zamanda JPA projesi istendiði
için object relational model olan hibernate tercih ettim. Daha önce kullanmiþ olduðum ve hakim olduðum içinde jsf tercihimi
primefacesden yana kullandim.


CREATE TABLE CONTACT (
    CONTACTID INT NOT NULL AUTO_INCREMENT,
    USERNAME VARCHAR(20) NOT NULL,
    USERSURNAME VARCHAR(20) NOT NULL,
    USEREMAIL VARCHAR(30) NOT NULL,
    USERADDRES VARCHAR(20) NOT NULL,
    PRIMARY KEY (CONTACTID)
);

CREATE TABLE CONTACTPHONE (
    CONTACTPHONEID INT NOT NULL AUTO_INCREMENT,
    CONTACTID INT NOT NULL,
    PHONENUMBER CHAR(11) NOT NULL,
    PRIMARY KEY (CONTACTPHONEID),
    CONSTRAINT FK_CONTACTPHONE FOREIGN KEY (CONTACTID)
        REFERENCES CONTACT (CONTACTID)
);
